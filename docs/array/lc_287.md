public int findRepeatNumber(int[] nums) {
    for (int i = 0; i < nums.length; i++) {
        while (nums[i] != i) {
            if (nums[i] == nums[nums[i]])
                return nums[i];
            int temp = nums[i];
            nums[i] = nums[nums[i]];
            nums[nums[i]] = temp;
        }
    }
    return -1;
}
